APP="/Applications/996传奇盒子.app"   # 改成实际路径

echo "==== 1. 架构 ===="
lipo -info "$APP/Contents/MacOS/"*
echo
echo "==== 2. 签名信息 ===="
codesign -dv --verbose=4 "$APP" 2>&1
echo
echo "==== 3. 签名完整性 ===="
codesign --verify --deep --strict --verbose=2 "$APP" 2>&1
echo
echo "==== 4. Gatekeeper 评估 ===="
spctl -a -vvv -t install "$APP" 2>&1
echo
echo "==== 5. 公证 staple ===="
xcrun stapler validate "$APP" 2>&1
echo
echo "==== 6. 隔离属性 ===="
xattr -l "$APP" 2>&1
echo
echo "==== 7. 系统日志（点开 app 后 5 秒抓） ===="
log show --predicate 'eventMessage contains "996传奇盒子" or eventMessage contains "AKInterface" or subsystem == "com.apple.syspolicy"' --info --last 5m 2>&1 | tail -50
